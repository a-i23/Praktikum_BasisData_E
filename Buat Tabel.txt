CREATE TABLE lokasi (
	id INT NOT NULL AUTO_INCREMENT,
	nama_lokasi VARCHAR (255) NOT NULL,
	PRIMARY KEY (id)
) ENGINE = InnoDB;

CREATE TABLE jabatan (
	id INT NOT NULL AUTO_INCREMENT,
	nama_jabatan VARCHAR (255) NOT NULL,
	gapok_jabatan DOUBLE NOT NULL,
	tunjangan_jabatan DOUBLE NOT NULL,
	uang_makan_perhari DOUBLE NOT NULL,
	PRIMARY KEY (id)
) ENGINE = InnoDB;

CREATE TABLE pengguna (
	id INT NOT NULL AUTO_INCREMENT,
	username VARCHAR(255) NOT NULL,
	password VARCHAR(255) NOT NULL,
	peran ENUM('ADMIN','USER'),
	login_terakhir TIMESTAMP NULL,
	PRIMARY KEY (id)
) ENGINE = InnoDB;

CREATE TABLE karyawan (
	id INT NOT NULL AUTO_INCREMENT,
	nik VARCHAR(255) NOT NULL,
	nama_lengkap VARCHAR(255) NOT NULL,
	handphone VARCHAR(255) NOT NULL,
	email VARCHAR(255) NOT NULL,
	tanggal_masuk DATE NOT NULL,
	pengguna_id INT,
	PRIMARY KEY (id)
) ENGINE = InnoDB;

CREATE TABLE bagian (
	id INT NOT NULL AUTO_INCREMENT ,
	nama_bagian VARCHAR(255) NOT NULL ,
	karyawan_id INT NOT NULL ,
	lokasi_id INT NOT NULL ,
	PRIMARY KEY (id)
) ENGINE = InnoDB;

CREATE TABLE jabatan_karyawan (
	id INT NOT NULL AUTO_INCREMENT ,
	jabatan_id INT NOT NULL ,
	karyawan_id INT NOT NULL ,
	tanggal_mulai DATE NOT NULL ,
	PRIMARY KEY (id)
) ENGINE = InnoDB;
